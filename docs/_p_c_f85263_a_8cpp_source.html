70void PCF85263A::set_alarm( int digit, int val, int int_sel )
71{
72 uint8_t en;
73
74 en = ((val & 80) ? 0x0 : 0x1) << digit;
75
76 reg_w( Second_alarm1 + digit, dec2bcd( val ) );
77 bit_op8( Alarm_enables, ~en, en );

79 uint8_t en_bit = (digit < 5) ? 0x10 : 0x80;
